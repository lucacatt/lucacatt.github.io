    <script type="text/javascript">
        function creaFigura() {
            mioCanvas = document.getElementById("myCanvas");
            ctx = mioCanvas.getContext("2d");

            ctx.beginPath();
            ctx.arc(100, 100, 50, 0, 2 * Math.PI, true);
            ctx.lineWidth = 3;
            ctx.strokeStyle = "blue";
            ctx.stroke();
            ctx.fillStyle = "white";

            ctx.beginPath();
            ctx.arc(210, 100, 50, 0, 2 * Math.PI, true);
            ctx.lineWidth = 3;
            ctx.strokeStyle = "black";
            ctx.stroke();
            ctx.fillStyle = "white";

            ctx.beginPath();
            ctx.arc(320, 100, 50, 0, 2 * Math.PI, true);
            ctx.lineWidth = 3;
            ctx.strokeStyle = "red";
            ctx.stroke();
            ctx.fillStyle = "white";

            ctx.beginPath();
            ctx.arc(150, 160, 50, 0, 2 * Math.PI, true);
            ctx.lineWidth = 3;
            ctx.strokeStyle = "yellow";
            ctx.stroke();
            ctx.fillStyle = "white";

            ctx.beginPath();
            ctx.arc(270, 160, 50, 0, 2 * Math.PI, true);
            ctx.lineWidth = 3;
            ctx.strokeStyle = "green";
            ctx.stroke();
            ctx.fillStyle = "white";
        }
    </script>
</head>



<body>

    <form>
        <input type="button" id="ok" value="Disegna" onclick="creaFigura()">
